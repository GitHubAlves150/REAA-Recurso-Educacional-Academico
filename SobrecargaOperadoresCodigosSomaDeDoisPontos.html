            #ifndef GEOMETRIA_H
            #define GEOMETRIA_H

            class Geometria
            {
                public:
                    Geometria(int=0, int=0, int=0);//cnt, pontos no espaço

                    Geometria operator+(Geometria);


                    int getX();//retorna o valor de x
                    int getY();//retorna o valor de y
                    int getZ();//retorna o valor de z


                private:
                    int x,
                        y,
                        z;

            };
            #endif//GEOMETRIA_H

            Arquivo .cpp
            
            #include 
            using namespace::std;
            #include "Geometria.hpp"
            
            Geometria::Geometria(int X, int Y, int Z):
                x(X), y(Y), z(Z)
            {
            //vazio
            }
            
            //Posso retornar um objeto também.
            //Pois objeto é um tipo de dados também.
            Geometria Geometria::operator+(Geometria P)
            {
                x=x+P.x;
                y=y+P.y;
                z=z+P.z;
                return Geometria(x, y, z);
            }
            
            int Geometria::getX()//retorna o valor de x
            {
                return x;
            }
            
            
            
            int Geometria::getY()//retorna o valor de y
            {
                return y;
            }
            
            int Geometria::getZ()//retorna o valor de z
            {
                return z;
            }
